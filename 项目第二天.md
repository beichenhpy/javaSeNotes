## 分页查询

```js
/*
                1.为了达到向百度一样的 显示十个 然后前五后四 前面不够5个 后面补齐10个 后面不足四个 前面补齐十个
                */
                //设计 begin 和 end
                //1.当总页数小于10时
                var begin;
                var end;
                if (data.totalPage < 10){
                    begin = 1;
                    end = data.totalPage;
                } else {
                    //当页数大于十
                    begin = data.currentPage - 5;
                    end = data.currentPage + 4;
                    //当前面不足5个时，即begin < 1
                    if (begin < 1){
                        //显示完10个
                        begin = 1;
                        end = begin + 9;
                    }
                    //到最后一页 后面不够4个时 即 end = totalPage
                    if (end > data.totalPage){
                        end = data.totalPage;
                        begin = end - 9;
                    }
                }
```
